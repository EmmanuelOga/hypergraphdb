434         * closure of this ontology
435         * 
436         * @param axiomType
437         *            The type of axiom to count
438         * @param includeImportsClosure
439         *            Specifies that the imports closure should be included when
440         *            counting axioms
441         * @return The number of the specified types of axioms in this ontology
442         */
443        public <T extends OWLAxiom> int getAxiomCount(AxiomType<T> axiomType,
444                        boolean includeImportsClosure) {
445                if (!includeImportsClosure) {
446                        return getAxiomCount(axiomType);
447                }
448                int result = 0;
449                for (OWLOntology ont : getImportsClosure()) {
450                        result += ont.getAxiomCount(axiomType);
451                }
452                return result;
453        }
